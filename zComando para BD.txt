CREATE TABLE usuarios (
    id NUMBER PRIMARY KEY,
    full_name VARCHAR2(50),
    email VARCHAR2(50),
    phone_number NUMBER,
    phone_extension NUMBER
);


CREATE TABLE tareas (
    id NUMBER PRIMARY KEY,
    name VARCHAR2(255),
    due_date DATE,
    priority VARCHAR2(10),
    status VARCHAR2(50),
    assigned_user_id NUMBER,
    FOREIGN KEY (assigned_user_id) REFERENCES usuarios(id)
);
